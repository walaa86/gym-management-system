}

Input: struct {M150, Walaa, 123-456, 39}

Process: Concatenate all fields with separator "#//#"

Output: "M150#//#Walaa#//#123-456#//#39"

Why "#//#"? This unique separator is unlikely to appear in actual data, preventing conflicts.

3 Write String Line to File

Now we have a text line. Let's write it to the file on disk.

SaveMemberDataLinetoFile() Function
void SaveMemberDataLinetoFile(string FileName, string strDataLine) {
    
    fstream MyFile;  // File stream object (tool for file operations)
    
    // Open file in output+append mode
    // ios::out = Output (write) mode
    // ios::app = Append mode (add to end, don't erase existing data)
    // Combined: Create file if doesn't exist, append if exists
    MyFile.open(FileName, ios::out | ios::app);
    
    // Check if file opened successfully
    if(MyFile.is_open()) {
        
        // Write the data line to file
        // endl adds newline character (\n) and flushes buffer
        MyFile << strDataLine << endl;
        
        // Close file to save changes and free system resources
        MyFile.close();
        
        // âœ… Data is now PERMANENTLY saved on disk!
    }
}

ğŸ“ File Opening Modes

Mode Symbol Behavior
Input ios::in Open for reading. File must exist.
Output ios::out Open for writing. Creates file or ERASES existing content.
Append ios::app Add to end of file. Preserves existing content.
Out+Append ios::out | ios::app Best for adding records: Creates file if missing, appends if exists.

vector<stMember> LoadMembersDataFromFile(string FileName) {
    
    vector<stMember> vMembers;  // Empty container for all members
    fstream MyFile;                // File stream object
    
    // Open file in input (read) mode
    MyFile.open(FileName, ios::in);
    
    // Check if file exists and opened successfully
    if(MyFile.is_open()) {
        
        string Line;       // Temporary variable to store each line
        stMember Member;  // Temporary variable for converted member
        
        // Read file line by line until end of file
        // getline() returns false when reaching EOF (End of File)
        while(getline(MyFile, Line)) {
            
            // Convert the string line to struct
            Member = ConvertLineToRecord(Line);
            
            // Add the member to our vector
            vMembers.push_back(Member);
        }
        
        // Close file to free system resources
        MyFile.close();
    }
    // If file doesn't exist, return empty vector
    
    return vMembers;  // Return all loaded members
}

SplitString() Function
vector<string> SplitString(string S1, string Delimiter) {
    
    vector<string> vString;  // Container for extracted fields
    short pos = 0;                // Position of delimiter in string
    string sWord;                // Temporary storage for extracted word
    
    // Loop until no more delimiters found
    while((pos = S1.find(Delimiter)) != std::string::npos) {
        
        // Extract word from start of string to delimiter position
        sWord = S1.substr(0, pos);
        
        // Add word to vector if not empty
        if(sWord != "") {
            vString.push_back(sWord);
        }
        
        // Remove processed part (word + delimiter) from string
        // This moves us forward to process the next word
        S1.erase(0, pos + Delimiter.length());
    }
    
    // Add the last remaining word (no delimiter after it)
    if(S1 != "") {
        vString.push_back(S1);
    }
    
    return vString;  // Return vector with all fields
}
