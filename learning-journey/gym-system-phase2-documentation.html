ğŸ“¤ Function 1: ConvertRecordToLine()

ğŸ¯ Purpose

Converts a stMember struct into a single text line with delimiters

âš™ï¸ Algorithm Steps

Step 1: Create an empty result string
Step 2: Append MemberId + Separator
Step 3: Append Name + Separator

ConvertRecordToLine.cpp Struct â†’ String Conversion 
// Function: Convert struct member to delimited string line
// Input: stMember object, Separator string (default: "#//#")
// Output: Single string line with all member data
// Example: "M150#//#Walaa Hussein#//#929-222-1111#//#39"

string ConvertRecordToLine(stMember Member, string Separator = "#//#"){
    
    string stMemberRecord = ""; // Initialize empty result string
    
    // Build the record line by concatenating fields with separator
    stMemberRecord += Member.MemberId + Separator;   // Add Member ID
    stMemberRecord += Member.Name + Separator;       // Add Name
    stMemberRecord += Member.Phone + Separator;      // Add Phone
    stMemberRecord += to_string(Member.Age);      // Add Age (convert int to string)
    
    return stMemberRecord; // Return the complete line
}

âœ‚ï¸ Function 2: SplitString()

ğŸ¯ Purpose

Splits a delimited string into individual fields stored in a vector

âš™ï¸ Algorithm Steps

Step 1: Create empty vector to store results
Step 2: Find position of first delimiter in string
Step 3: If delimiter found:
  • Extract substring from start to delimiter position
  • Add extracted word to vector (if not empty)
  • Remove processed part from original string
  • Repeat from Step 2
Step 4: When no more delimiters found, add remaining string as last word
Step 5: Return vector containing all extracted fields
SplitString.cpp String Parsing Engine 
// Function: Split a delimited string into individual components
// Input: String to split, Delimiter to search for
// Output: Vector of strings containing all extracted fields
// Example: "A#//#B#//#C" â†’ ["A", "B", "C"]

vector<string> SplitString(string S1, string Delimiter){
    
    vector<string> vString; // Container for extracted words
    
    short pos = 0;      // Position of delimiter in string
    string sWord;        // Temporary variable to store extracted word
    
    // Loop until no more delimiters are found
    while ((pos = S1.find(Delimiter)) != std::string::npos){
        
        // Extract word from start of string to delimiter position
        sWord = S1.substr(0, pos);
        
        // Add word to vector if it's not empty
        if (sWord != ""){
            vString.push_back(sWord);
        }
        
        // Remove processed part (word + delimiter) from string
        // This moves us forward to process the next word
        S1.erase(0, pos + Delimiter.length()); 
    }
    
    // Add the last remaining word (no delimiter after it)
    if (S1 != ""){
        vString.push_back(S1);
    }
    
    return vString; // Return vector with all extracted fields
}

ğŸ“¥ Function 3: ConvertLineToRecord()

ğŸ¯ Purpose

Converts a delimited string line back into a stMember struct

âš™ï¸ Algorithm Steps

Step 1: Create empty Member struct and vector
Step 2: Call SplitString() to parse the line into fields
Step 3: Assign vector[0] to Member.MemberId
Step 4: Assign vector[1] to Member.Name

Step 6: Convert vector[3] from string to int and assign to Member.Age

ConvertLineToRecord.cpp String â†’ Struct Conversion 
// Function: Convert delimited string line back to struct
// Input: String line with delimited data, Separator string
// Output: stMember struct with all fields populated
// Example: "M150#//#Walaa#//#123#//#39" â†’ stMember object

stMember ConvertLineToRecord(string Line, string Separator = "#//#"){
    
    stMember Member;                    // Create empty member struct
    vector<string> vMemberData;       // Vector to hold parsed data
    
    // Parse the line into individual fields
    vMemberData = SplitString(Line, Separator);
    
    // Assign each field from vector to corresponding struct member
    Member.MemberId = vMemberData[0];          // First field: Member ID
    Member.Name = vMemberData[1];              // Second field: Name
    Member.Phone = vMemberData[2];             // Third field: Phone
    Member.Age = stoi(vMemberData[3]);      // Fourth field: Age (convert string to int)
    
    return Member; // Return the reconstructed struct
}

// Function: Split delimited string into individual fields
// Input: String to parse, delimiter to search for
// Output: Vector of extracted string fields
// Algorithm: Iteratively find, extract, and remove fields
vector<string> SplitString(string S1, string Delimiter){
    
    vector<string> vString;
    short pos = 0;
    string sWord;
    
    // Loop: find delimiter, extract word, remove processed part
    while((pos = S1.find(Delimiter)) != std::string::npos){
        
        // Extract substring from beginning to delimiter
        sWord = S1.substr(0, pos);
        
        // Add to vector if not empty
        if(sWord != ""){
            vString.push_back(sWord);
        }
        
        // Remove processed portion (word + delimiter)
        S1.erase(0, pos + Delimiter.length());
    }
    
    // Add the final remaining word (no delimiter after it)
    if(S1 != ""){
        vString.push_back(S1);
    }
    
    return vString;
}
