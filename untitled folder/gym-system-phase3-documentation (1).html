/*
 * Function: Save a single member record line to file
 * Input: 
 *   - FileName: Path to the file (e.g., "Members.txt")
 *   - strDataLine: Pre-formatted member data line
 * Output: None (writes to file)
 * Mode: Append (ios::out | ios::app) - adds to end without erasing
 */

void SaveMemberDataLinetoFile(string FileName, string strDataLine){
    
    fstream MyFile;  // File stream object for read/write operations
    
    // Open file in output+append mode
    // ios::out  - Output (write) mode
    // ios::app  - Append mode (add to end, don't erase existing)
    // Together: Create file if doesn't exist, append if exists
    MyFile.open(FileName, ios::out | ios::app);
    
    // Check if file opened successfully
    if(MyFile.is_open()){
        
        // Write the data line to file
        // endl adds newline character and flushes buffer
        MyFile << strDataLine << endl;
        
        // Close file to save changes and free resources
        MyFile.close();
    }
    // Note: If file fails to open, function silently fails
    // Production code should include error handling here
}

ğŸ“‚ Function 4: LoadMembersDataFromFile()

ğŸ¯ Purpose

Read all member records from file into memory (vector)

âš™ï¸ Algorithm Steps

Step 1: Create empty vector to store members
Step 2: Create fstream object and open file in read mode (ios::in)
Step 3: Check if file opened successfully

LoadMembersDataFromFile.cpp File â†’ Memory Loading 
/*
 * Function: Load all members from file into vector
 * Input: FileName - Path to the members file
 * Output: vector - All members from file
 * Process: Read line by line, convert each to struct, add to vector
 */

vector<stMember> LoadMembersDataFromFile(string FileName){
    
    vector<stMember> vMember;  // Container for all members
    
    fstream MyFile;  // File stream object
    
    // Open file in input (read) mode
    MyFile.open(FileName, ios::in);
    
    // Check if file exists and opened successfully
    if(MyFile.is_open()){
        
        string Line;       // Temporary storage for each line
        stMember Member;  // Temporary storage for converted member
        
        // Read file line by line until end of file
        // getline() returns false when reaching end of file
        while(getline(MyFile, Line)){
            
            // Convert the delimited string line to struct
            Member = ConvertLineToRecord(Line);
            
            // Add the member to our vector
            vMember.push_back(Member);
        }
        
        // Close file to free resources
        MyFile.close();
    }
    // If file doesn't exist or can't open, return empty vector
    
    return vMember;  // Return all loaded members
}

vector<string> SplitString(string S1, string Delimiter){
    vector<string> vString;
    short pos = 0;
    string sWord;
    
    while((pos = S1.find(Delimiter)) != std::string::npos){
        sWord = S1.substr(0, pos);
        if(sWord != ""){
            vString.push_back(sWord);
        }
        S1.erase(0, pos + Delimiter.length());
    }
    
    if(S1 != ""){
        vString.push_back(S1);
    }
    
    return vString;
}

stMember ConvertLineToRecord(string Line, string Separator = "#//#"){
    stMember Member;
    vector<string> vMemberData;
    
    vMemberData = SplitString(Line, Separator);
    
    Member.MemberId = vMemberData[0];
    Member.Name = vMemberData[1];
    Member.Phone = vMemberData[2];
    Member.Age = stoi(vMemberData[3]);
    
    return Member;
}
